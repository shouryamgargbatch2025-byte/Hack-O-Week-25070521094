  <script>
    let books = [];

    const capacityInput = document.getElementById("capacity");
    const titleInput = document.getElementById("title");
    const authorInput = document.getElementById("author");
    const categoryInput = document.getElementById("category");
    const removeTitleInput = document.getElementById("removeTitle");
    const searchInput = document.getElementById("searchInput");
    const statusBox = document.getElementById("statusBox");
    const bookList = document.getElementById("bookList");

    function getCapacity() {
      return Math.max(1, Number(capacityInput.value) || 1);
    }

    function setStatus(message, kind = "neutral") {
      const colors = {
        neutral: "#f8fafc",
        success: "#ecfdf5",
        error: "#fef2f2"
      };
      statusBox.style.background = colors[kind];
      statusBox.textContent = message;
    }

    function renderBooks() {
      const searchText = searchInput.value.trim().toLowerCase();
      const filteredBooks = books.filter((book) => book.title.toLowerCase().includes(searchText));

      document.getElementById("countStat").textContent = books.length;
      document.getElementById("capacityStat").textContent = getCapacity();
      document.getElementById("remainingStat").textContent = Math.max(0, getCapacity() - books.length);

      if (!filteredBooks.length) {
        bookList.innerHTML = `<div class="empty">${searchText ? "No matching books found." : "No books in the catalogue yet."}</div>`;
        return;
      }

      bookList.innerHTML = filteredBooks.map((book, index) => `
        <article class="book">
          <span class="pill">Book ${index + 1}</span>
          <h3>${book.title}</h3>
          <div class="meta">Author: ${book.author}</div>
          <div class="meta">Category: ${book.category}</div>
        </article>
      `).join("");
    }

    function addBook() {
      const capacity = getCapacity();
      const title = titleInput.value.trim();
      const author = authorInput.value.trim();
      const category = categoryInput.value.trim();

      if (!title || !author || !category) {
        setStatus("Please fill in title, author, and category before adding a book.", "error");
        return;
      }

      if (books.length >= capacity) {
        setStatus(`Cannot add more books. Capacity limit of ${capacity} has been reached.`, "error");
        return;
      }

      books.push({ title, author, category });
      setStatus(`"${title}" was added successfully.`, "success");
      renderBooks();
    }

    function removeBook() {
      const target = removeTitleInput.value.trim().toLowerCase();
      if (!target) {
        setStatus("Enter the exact title you want to remove.", "error");
        return;
      }

      const originalLength = books.length;
      books = books.filter((book) => book.title.toLowerCase() !== target);

      if (books.length === originalLength) {
        setStatus("No book matched that title, so nothing was removed.", "error");
        return;
      }

      setStatus("Book removed successfully.", "success");
      removeTitleInput.value = "";
      renderBooks();
    }

    function loadSampleData() {
      const samples = [
        { title: "Clean Code", author: "Robert C. Martin", category: "Software" },
        { title: "The Pragmatic Programmer", author: "Andrew Hunt", category: "Programming" },
        { title: "You Don't Know JS", author: "Kyle Simpson", category: "JavaScript" }
      ];

      books = [];
      const capacity = getCapacity();
      samples.slice(0, capacity).forEach((book) => books.push(book));
      setStatus("Sample books loaded into the catalogue.", "success");
      renderBooks();
    }

    function syncCapacity() {
      const capacity = getCapacity();
      if (books.length > capacity) {
        books = books.slice(0, capacity);
        setStatus(`Capacity reduced to ${capacity}. Extra books were trimmed to fit the limit.`, "error");
      } else {
        setStatus(`Capacity updated to ${capacity}.`, "neutral");
      }
      renderBooks();
    }

    document.getElementById("addBtn").addEventListener("click", addBook);
    document.getElementById("removeBtn").addEventListener("click", removeBook);
    document.getElementById("loadSampleBtn").addEventListener("click", loadSampleData);
    document.getElementById("clearSearchBtn").addEventListener("click", () => {
      searchInput.value = "";
      renderBooks();
    });
    searchInput.addEventListener("input", renderBooks);
    capacityInput.addEventListener("input", syncCapacity);

    renderBooks();
  </script>
