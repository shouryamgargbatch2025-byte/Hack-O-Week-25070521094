<body>
  <div class="box">
    <h2>Printer Queue</h2>
    <input type="text" id="jobInput" placeholder="Enter Print Job Name" />
    <button onclick="enqueueJob()">Add Print Job</button>
    <button onclick="dequeueJob()">Print Next Job</button>

    <div class="preview">
      <h3>Queue Preview</h3>
      <div id="queueList">No jobs in queue</div>
      <h3>Last Printed:</h3>
      <p id="printedJob">None</p>
    </div>
  </div>

  <script>
    let queue = [];

    function enqueueJob() {
      const job = document.getElementById("jobInput").value.trim();
      if (!job) {
        alert("Enter a print job");
        return;
      }

      queue.push(job);
      document.getElementById("jobInput").value = "";
      updateQueue();
    }

    function dequeueJob() {
      if (queue.length === 0) {
        alert("No jobs in queue");
        return;
      }

      const printed = queue.shift();
      document.getElementById("printedJob").innerText = printed;
      updateQueue();
    }

    function updateQueue() {
      const queueList = document.getElementById("queueList");
      if (queue.length === 0) {
        queueList.innerHTML = "No jobs in queue";
      } else {
        queueList.innerHTML = queue.map((job, index) =>
          `<div class="job">${index + 1}. ${job}</div>`
        ).join("");
      }
    }
  </script>
