    <div class="preview" id="treePreview">Root</div>

  <script>
    class Node {
      constructor(value) {
        this.value = value;
        this.left = null;
        this.right = null;
      }
    }

    let root = new Node("Root");

    function insertNode(node, value) {
      if (value.localeCompare(node.value) < 0) {
        if (!node.left) node.left = new Node(value);
        else insertNode(node.left, value);
      } else {
        if (!node.right) node.right = new Node(value);
        else insertNode(node.right, value);
      }
    }

    function insertFolder() {
      const folder = document.getElementById("folderName").value.trim();
      if (!folder) {
        alert("Please enter a folder name");
        return;
      }

      insertNode(root, folder);
      document.getElementById("folderName").value = "";
      renderTree();
    }

    function preorder(node, arr = []) {
      if (!node) return arr;
      arr.push(node.value);
      preorder(node.left, arr);
      preorder(node.right, arr);
      return arr;
    }

    function inorder(node, arr = []) {
      if (!node) return arr;
      inorder(node.left, arr);
      arr.push(node.value);
      inorder(node.right, arr);
      return arr;
    }

    function postorder(node, arr = []) {
      if (!node) return arr;
      postorder(node.left, arr);
      postorder(node.right, arr);
      arr.push(node.value);
      return arr;
    }

    function treeText(node, level = 0) {
      if (!node) return "";
      let text = "";
      text += treeText(node.right, level + 1);
      text += " ".repeat(level * 4) + node.value + "\n";
      text += treeText(node.left, level + 1);
      return text;
    }

    function renderTree() {
      document.getElementById("treePreview").textContent = treeText(root);
    }

    function showPreorder() {
      document.getElementById("traversalOutput").textContent =
        "Preorder: " + preorder(root).join(", ");
    }

    function showInorder() {
      document.getElementById("traversalOutput").textContent =
        "Inorder: " + inorder(root).join(", ");
    }

    function showPostorder() {
      document.getElementById("traversalOutput").textContent =
        "Postorder: " + postorder(root).join(", ");
    }

    renderTree();
  </script>
