<body>
  <div class="container">
    <h2>Bank ATM Queue</h2>
    <input type="text" id="customerName" placeholder="Enter customer name" />
    <input type="text" id="tokenNo" placeholder="Enter token number" />
    <button onclick="enqueueCustomer()">Add Customer</button>
    <button onclick="serveCustomer()">Serve Next Customer</button>

    <div id="queueList"></div>
    <div class="status" id="queueStatus">Front: - | Rear: -</div>
  </div>

  <script>
    let queue = [];
    const capacity = 8;

    function enqueueCustomer() {
      const name = document.getElementById("customerName").value.trim();
      const token = document.getElementById("tokenNo").value.trim();

      if (!name || !token) {
        alert("Please enter customer name and token number");
        return;
      }

      if (queue.length >= capacity) {
        alert("Queue is full");
        return;
      }

      queue.push({ name, token });
      document.getElementById("customerName").value = "";
      document.getElementById("tokenNo").value = "";
      renderQueue();
    }

    function serveCustomer() {
      if (queue.length === 0) {
        alert("Queue is empty");
        return;
      }

      queue.shift();
      renderQueue();
    }

    function renderQueue() {
      const queueList = document.getElementById("queueList");
      queueList.innerHTML = queue.length
        ? queue.map(c => `<div class="customer">Token ${c.token} - ${c.name}</div>`).join("")
        : "<p>Queue is empty</p>";

      const front = queue.length ? queue[0].token : "-";
      const rear = queue.length ? queue[queue.length - 1].token : "-";

      document.getElementById("queueStatus").textContent = `Front: ${front} | Rear: ${rear}`;
    }
  </script>
