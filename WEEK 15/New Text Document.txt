<body>
  <div class="box">
    <h2>Contact Book Search</h2>
    <input type="text" id="contactName" placeholder="Enter Contact Name" />
    <input type="text" id="contactPhone" placeholder="Enter Phone Number" />
    <button onclick="addContact()">Add Contact</button>

    <input type="text" id="searchName" placeholder="Search by Name" />
    <button onclick="searchContact()">Search Contact</button>

    <div class="preview">
      <h3>All Contacts</h3>
      <div id="contactList">No contacts added</div>

      <h3>Search Result</h3>
      <p id="searchResult">No search yet</p>
    </div>
  </div>

  <script>
    let contacts = [];

    function addContact() {
      const name = document.getElementById("contactName").value.trim();
      const phone = document.getElementById("contactPhone").value.trim();

      if (!name || !phone) {
        alert("Please enter both name and phone number");
        return;
      }

      contacts.push({ name, phone });
      contacts.sort((a, b) => a.name.localeCompare(b.name));
      updateContacts();

      document.getElementById("contactName").value = "";
      document.getElementById("contactPhone").value = "";
    }

    function binarySearch(name) {
      let left = 0;
      let right = contacts.length - 1;

      while (left <= right) {
        let mid = Math.floor((left + right) / 2);
        let compare = contacts[mid].name.toLowerCase().localeCompare(name.toLowerCase());

        if (compare === 0) return contacts[mid];
        if (compare < 0) left = mid + 1;
        else right = mid - 1;
      }

      return null;
    }

    function searchContact() {
      const name = document.getElementById("searchName").value.trim();
      const result = binarySearch(name);

      if (result) {
        document.getElementById("searchResult").innerText =
          `Name: ${result.name}, Phone: ${result.phone}`;
      } else {
        document.getElementById("searchResult").innerText = "Contact not found";
      }
    }

    function updateContacts() {
      const contactList = document.getElementById("contactList");

      if (contacts.length === 0) {
        contactList.innerHTML = "No contacts added";
      } else {
        contactList.innerHTML = contacts.map(contact =>
          `<div class="contact">${contact.name} - ${contact.phone}</div>`
        ).join("");
      }
    }
  </script>
