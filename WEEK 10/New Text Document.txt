  <script>
    const ids = [
      "institute",
      "courseTitle",
      "courseTagline",
      "description",
      "duration",
      "mode",
      "ctaText",
      "featureInput"
    ];

    const elements = Object.fromEntries(ids.map((id) => [id, document.getElementById(id)]));

    function buildFeatures(value) {
      const features = value.split(",").map((item) => item.trim()).filter(Boolean);
      const fallback = ["HTML Foundations", "CSS Styling", "JavaScript Logic", "Responsive Design"];
      return (features.length ? features : fallback).slice(0, 8);
    }

    function updatePreview() {
      document.getElementById("previewInstitute").textContent = elements.institute.value || "Institute Name";
      document.getElementById("previewTitle").textContent = elements.courseTitle.value || "Course Title";
      document.getElementById("previewTagline").textContent = elements.courseTagline.value || "Course Tagline";
      document.getElementById("previewDescription").textContent = elements.description.value || "Course description";
      document.getElementById("previewDuration").textContent = elements.duration.value || "Duration";
      document.getElementById("previewMode").textContent = elements.mode.value || "Mode";
      document.getElementById("previewButton").textContent = elements.ctaText.value || "Join Now";

      const featureMarkup = buildFeatures(elements.featureInput.value).map((feature, index) => `
        <article class="feature">
          <span>${index + 1}</span>
          <strong>${feature}</strong>
          <p>Practical lessons and guided exercises focused on ${feature.toLowerCase()}.</p>
        </article>
      `).join("");

      document.getElementById("featuresList").innerHTML = featureMarkup;
    }

    ids.forEach((id) => elements[id].addEventListener("input", updatePreview));
    updatePreview();
  </script>
