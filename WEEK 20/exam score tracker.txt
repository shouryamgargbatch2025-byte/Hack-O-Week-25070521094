<body>
  <div class="container">
    <h2>Exam Score Tracker</h2>
    <input type="number" id="scoreInput" placeholder="Enter exam score" />
    <button onclick="addScore()">Add Score</button>

    <div id="scoreList">No scores added</div>
    <div class="stats" id="scoreStats">Min: - | Max: - | Mode: -</div>
  </div>

  <script>
    let scores = [];

    function addScore() {
      const score = parseInt(document.getElementById("scoreInput").value);

      if (isNaN(score)) {
        alert("Please enter a valid score");
        return;
      }

      scores.push(score);
      document.getElementById("scoreInput").value = "";
      renderScores();
    }

    function getMode(arr) {
      let freq = {};
      let maxCount = 0;
      let mode = "-";

      for (let num of arr) {
        freq[num] = (freq[num] || 0) + 1;
        if (freq[num] > maxCount) {
          maxCount = freq[num];
          mode = num;
        }
      }

      return mode;
    }

    function renderScores() {
      const scoreList = document.getElementById("scoreList");

      if (scores.length === 0) {
        scoreList.innerHTML = "No scores added";
        document.getElementById("scoreStats").textContent = "Min: - | Max: - | Mode: -";
        return;
      }

      scoreList.innerHTML = scores.map(score => `<div class="score">${score}</div>`).join("");

      const min = Math.min(...scores);
      const max = Math.max(...scores);
      const mode = getMode(scores);

      document.getElementById("scoreStats").textContent =
        `Min: ${min} | Max: ${max} | Mode: ${mode}`;
    }
  </script>
