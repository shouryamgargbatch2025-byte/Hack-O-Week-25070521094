<body>
  <div class="box">
    <h2>Browser History</h2>
    <input type="text" id="urlInput" placeholder="Enter URL" />
    <button onclick="visitPage()">Visit Page</button>
    <button onclick="goBack()">Back</button>
    <button onclick="goForward()">Forward</button>

    <div class="preview">
      <h3>Current Page:</h3>
      <p id="currentPage">No page visited yet</p>

      <h3>Back Stack:</h3>
      <p id="backHistory">Empty</p>

      <h3>Forward Stack:</h3>
      <p id="forwardHistory">Empty</p>
    </div>
  </div>

  <script>
    let backStack = [];
    let forwardStack = [];
    let currentPage = "";

    function visitPage() {
      const url = document.getElementById("urlInput").value.trim();
      if (!url) {
        alert("Enter a URL");
        return;
      }

      if (currentPage) {
        backStack.push(currentPage);
      }

      currentPage = url;
      forwardStack = [];
      updatePreview();
      document.getElementById("urlInput").value = "";
    }

    function goBack() {
      if (backStack.length === 0) {
        alert("No back history");
        return;
      }

      forwardStack.push(currentPage);
      currentPage = backStack.pop();
      updatePreview();
    }

    function goForward() {
      if (forwardStack.length === 0) {
        alert("No forward history");
        return;
      }

      backStack.push(currentPage);
      currentPage = forwardStack.pop();
      updatePreview();
    }

    function updatePreview() {
      document.getElementById("currentPage").innerText = currentPage || "No page visited yet";
      document.getElementById("backHistory").innerText = backStack.length ? backStack.join(" -> ") : "Empty";
      document.getElementById("forwardHistory").innerText = forwardStack.length ? forwardStack.join(" -> ") : "Empty";
    }
  </script>
