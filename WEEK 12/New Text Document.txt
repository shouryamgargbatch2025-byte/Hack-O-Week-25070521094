<body>
  <div class="box">
    <h2>Shopping Cart System</h2>
    <input type="text" id="itemId" placeholder="Enter Item ID" />
    <input type="text" id="itemName" placeholder="Enter Item Name" />
    <input type="number" id="itemPrice" placeholder="Enter Item Price" />
    <select id="position">
      <option value="tail">Insert at Tail</option>
      <option value="head">Insert at Head</option>
    </select>
    <button onclick="addItem()">Add Item</button>
    <input type="text" id="removeId" placeholder="Enter Item ID to Remove" />
    <button onclick="removeItem()">Remove Item</button>

    <div class="preview">
      <h3>Cart Preview</h3>
      <div id="cartList"></div>
      <h3 id="totalPrice">Total: Rs. 0</h3>
    </div>
  </div>

  <script>
    let cart = [];

    function addItem() {
      const id = document.getElementById("itemId").value.trim();
      const name = document.getElementById("itemName").value.trim();
      const price = parseFloat(document.getElementById("itemPrice").value);
      const position = document.getElementById("position").value;

      if (!id || !name || isNaN(price)) {
        alert("Please enter valid item details.");
        return;
      }

      const item = { id, name, price };

      if (position === "head") {
        cart.unshift(item);
      } else {
        cart.push(item);
      }

      displayCart();
      document.getElementById("itemId").value = "";
      document.getElementById("itemName").value = "";
      document.getElementById("itemPrice").value = "";
    }

    function removeItem() {
      const removeId = document.getElementById("removeId").value.trim();
      cart = cart.filter(item => item.id !== removeId);
      displayCart();
      document.getElementById("removeId").value = "";
    }

    function displayCart() {
      const cartList = document.getElementById("cartList");
      const totalPrice = document.getElementById("totalPrice");
      cartList.innerHTML = "";

      let total = 0;
      cart.forEach(item => {
        total += item.price;
        cartList.innerHTML += `<div class="item">ID: ${item.id} | ${item.name} | Rs. ${item.price}</div>`;
      });

      totalPrice.innerText = `Total: Rs. ${total}`;
    }
  </script>
